\sld{What is Stan?}
%
\begin{itemize}
\item Stan is an \myemph{imperative probabilistic programming language}
  \begin{subitemize}
  \item  cf., BUGS: declarative; \ Church, JAX: functional; PyMC/NumPyro: mix
  \end{subitemize}
\item Stan \myemph{program}: defines a probability model
  \begin{subitemize}
  \item declares data and (constrained) parameter variables
  \item defines log posterior (or penalized likelihood)
  \item defines posterior predictive quantities (e.g., predictions, event probabilities)
  \end{subitemize}
\item Stan \myemph{inference}: fits model to data \& makes predictions
  \begin{subitemize}
  \item MCMC for full Bayesian inference
  \item VB for approximate Bayesian inference
  \item MLE for penalized maximum likelihood estimation
  \end{subitemize}
\end{itemize}

\sld{The central limit theorem}
%
\vspace*{-3pt}
\begin{center}
\vspace*{-3pt}
\includegraphics[width=0.4\textwidth]{img/coin-flip-clt.png}

\begin{subitemize}
\item \myemph{proportion heads} for \myemph{100 sequences} (light grey lines) of 100,000 flips
\vspace*{-2pt}
\item \myemph{converges to expected value} 0.5 (red), with \myemph{standard deviation} $0.5 / \sqrt{\textsf{\# draws}}$
  \begin{subitemize}
  \item single flip standard deviation is 0.5
  \item standard deviation is \myemph{standard error of maximum likelihood estimate}, $\widehat{\theta} = \textsf{\# heads} / \textsf{\# flips}$
  \end{subitemize}

\sld{The central limit theorem (cont.)}
%
\begin{itemize}
\item \myemph{The} theorem of statistics
\vspace*{-4pt}
\begin{subsubitemize}
\item Cardano (1501--1576) conjectured convergence; (Jacob) Bernoulli
  (1713) proved convergence for binomials (law of large numbers); de Moivre
  (1733) conjectured the CLT; Laplace (1812) proved i.i.d.\ version;
  Lyapunov (1901) removed i.i.d.\ constraint
\end{subsubitemize}
\item Sample \myemph{mean} of $N$ independent variables with finite expectation/variance
\begin{subitemize}
\item  \myemph{converges} to their expected mean as $N \rightarrow \infty$
\item relevant for \myemph{sampling-based Bayesian inference} as provided by Stan 
\item \myemph{rate} of convergence is {\large ${\mathcal{O}}\!\left( \frac{1}{\sqrt{N}} \right)$}
  with factor for standard deviation
\item i.e., each \myemph{decimal place of accuracy} requires \myemph{$100 \times$ more draws}
\end{subitemize}
\end{itemize}

\sld{Random variables \large (a la Gelman and Hill)}
%
\begin{itemize}
\item Imagine an urn filled with balls, each \myemph{recording the numerical value of every random variable}.
  \begin{subitemize}
  \item A random variable is a mapping from balls to values.
  \item One ball represents the \myemph{real world}, but we don't know which.
  \end{subitemize}
\vfill
\item Or imagine \myemph{seeding a pseudorandom number generator}.
  \begin{subitemize}
  \item Given a \myemph{fixed seed}, the \myemph{sequence of pseudorandom numbers is determinstic}.
  \item A pseudorandom variable is a mapping from seeds to values, e.g., \texttt{rnorm()} in R.
  \end{subitemize}
\vfill
\end{itemize}


\sld{Examples of random variables}    
\begin{itemize}
\item the sum of the three best among a roll of four six-sided dice
\item the time before the next traffic accident on a given highway
\item the prevalence of a disease in a population
\item the ability of a student to answer exam questions
\vfill
\item \myemph{Frequentist statistics} only uses random variables for \myemph{observable quantities} in a sequence, with probability defined as long-term sequence probability.
\item \myemph{Bayesian statistics} uses random variables for \myemph{unobservable latent values, too.}
\end{itemize}

\sld{Events}
%
\begin{itemize}
\item An \myemph{event} is a something that can occur or not occur.
\item Events the main \myemph{bearers of probability} in statistics.
\item An event is defined by \myemph{a property of random variables} (i.e., a subset of the balls in the urn), e.g., 
\begin{subitemize}
\item the prevalence of disease is greater than 2\%, i.e., $\lambda > 0.02$
\item education scheme one is better than scheme two, i.e., $\theta_1 >
  \theta_2$
\end{subitemize}
\item The \myemph{probabilty of an event} is the probabiltiy that the random
  variables take on values that satisfy the property defining the event,
  \begin{subitemize}
  \item for the event $\lambda > 0.02$, the value $\Pr[\lambda > 0.02]$ is the probability
    that the random variable $\lambda$ takes on a value greater than 0.02 (i.e., the probability that a random ball choosen from the urn has $\lambda > 0.02$).
  \end{subitemize}
\end{itemize}

\sld{Bernoulli distribution}
%
\begin{itemize}
\item The \myemph{Bernoulli distribution} has a binary outcome $y \in \{ 0, 1 \}$,
  with 1 conventionally denoting ``success.''
\item The probability $y$ is 1, written $\Pr[y = 1]$, is defined to be $\theta \in (0,1)$, so
$$
  \distro{Bernoulli}(y \mid \theta)
  = \begin{cases}
    \theta & \textsf{if } y = 1,
    \\
    1 - \theta & \textsf{if } y = 0, \textsf{ and}
    \\
    0 & \textsf{otherwise}.
    \end{cases}
    $$
  \item For example, if $y \in \{ 0, 1 \}$ is a fair coin flip, we write
    $$
    y \sim \distro{Bernoulli}(0.5)
    $$
    to indicate that the probability mass function of $y$ is $\distro{Bernoulli}(y \mid 0.5)$,
    which follows from $\Pr[y = 1] = 0.5$.
\end{itemize}


\sld{Binomial distribution}
%
\begin{itemize}
\item In $\distro{Binomial}(y \mid N, \theta)$,
  $y$ is the \myemph{number of successes} in $N$ \myemph{independent trials}, where each trial has
  a $\theta \in (0, 1)$ \myemph{chance of success}.
\item If $y_1,\ldots,y_N\sim \distro{Bernoulli}(\theta)$,
then $(y_1 + \cdots + y_N) \sim \distro{Binomial}(N,\theta)$.
\item The \myemph{analytic form} is
\[
\distro{Binomial}(y \mid N,\theta)
\ = \ \binom{N}{y} \cdot \theta^{y} \cdot (1 - \theta)^{N-y},
\]
where \myemph{binomial coefficient} $\binom{N}{y}$ is the \myemph{number of sequences} of 0s and 1s of length $N$ with exactly $y$ 1s.  
\end{itemize}



\sld{Beta distribution}
%
\begin{itemize}
\item \myemph{Beta distributions} are used to \myemph{model probabilities} or proportions.
\item For parameters $\alpha, \beta > 0$ and a \myemph{continuous outcome} $\theta \in (0,1)$, 
\[
\distro{Beta}(\theta \mid \alpha,\beta) 
\ = \ \frac{1}{\Betafun(\alpha,\beta)}
      \cdot \theta^{\alpha - 1} 
      \cdot (1 - \theta)^{\beta-1}
\]
\item Euler's \myemph{Beta function} $\Betafun(\alpha, \beta)$ is required to \myemph{normalize}.
\item \myemph{Uniform is a beta}:
  $\distro{Uniform}(\theta\mid (0,1)) = \distro{Beta}(\theta \mid 1, 1) = $
\item \myemph{Conujugacy} (Laplace): With a beta prior and binomial data generating
  distribution, the posterior is a beta distribution.
\end{itemize}


\sld{Bayes's binomial model}
%
\begin{itemize}
\item \myemph{Data}
\begin{subitemize}
\item $y$: total \myemph{number of male live births} (251,527)
\item $N$ : total \myemph{number of live births} (493,472)
\end{subitemize}
\item \myemph{Parameter}: 
$\theta \in (0,1)$ for the \myemph{proportion of male} live births
\item \myemph{Prior}: $\theta \sim \distro{Beta}(1, 1)$
\item \myemph{Data generating distribution}: $y \sim \distro{Binomial}(N, \theta)$
\item \myemph{Joint density}: $p(y, \theta \mid N) = \distro{Binomial}(y \mid N, \theta)
  \cdot \distro{Beta}(\theta \mid 1, 1)$
\item \myemph{Posterior density}: $p(\theta \mid y) = \distro{Beta}(\theta \mid y + 1, N - y + 1)$
\end{itemize}

\sld{Estimation}
%
\begin{itemize}
\item \myemph{Posterior} is $\distro{Beta}(\theta \, | \, 1 + 241\,945, \ 1 + 251\,527)$
\item Posterior \myemph{mean}:
\[
\frac{1 + 251\,527}
     {1 + 241\,945 + 1 + 251\,527}
\approx 0.509708\underbrace{\bf 73}
\]
\item With uniform prior, \myemph{maximum likelihood estimate} is the \myemph{posterior mode},
\[
\frac{251\,527}
     {241\,945 + 251\,527}
\approx 0.509708\underbrace{\bf 82}.
\]
\item As number of observations approaches $\infty$, posterior mean approaches MLE.
\end{itemize}


\sld{Event probability}
%
\begin{itemize}
\item What is probability that a male live birth is more probable?
\begin{eqnarray*}
\Prob{\theta > 0.5}
& = &  \int_{\Theta} \indicator{\theta > 0.5} \, p(\theta \mid y,N) \textrm{d}\theta
\\[4pt]
  & \approx &  1 - 10^{-42}
  \\[4pt]
  & = & \textsf{\footnotesize 0.9999999999 9999999999 9999999999 9999999999 99}.
\end{eqnarray*}
\item $\indicator{\phi} = 1$ if condition $\phi$ is true and 0 otherwise. 
\item  Laplace solved Bayes's integral by
\begin{subitemize}
\item determining that the posterior was a beta distribution (conjugacy!)
\item and solving the normalization (gamma functions)
\end{subitemize}
\end{itemize}

\sld{Bayesian data analysis}
%
\begin{itemize}
\item ``By {Bayesian data analysis}, we mean {practical methods}
  for making {inferences} from {data} using {probability models}
  for quantities we {observe} and about which we {wish to learn}.''
  %
\item ``The essential characteristic of Bayesian methods is
  their \myemph{explicit use of probability for quantifying uncertainty}
  in inferences based on statistical analysis.''
\end{itemize}
%
\vfill\hfill{\footnotesize Gelman et al., {\slshape Bayesian Data Analysis},
  3rd edition, 2013}


\sld{Bayesian workflow}
%
\begin{itemize}
\item Set up \myemph{full probability model}
  \vspace*{-4pt}
  \begin{itemize}
  \item for \myemph{all} observable \& unobservable quantities
  \item consistent w. problem knowledge \& data collection
  \end{itemize}
  %
\item \myemph{Condition} on observed data (where Stan comes in)
  \vspace*{-4pt}
  \begin{itemize}
  \item to calculate posterior probability of unobserved quantities
    (e.g., parameters, predictions, missing data)
  \end{itemize}
  %
\item \myemph{Evaluate} (where Stan sticks around)
  \vspace*{-4pt}
  \begin{itemize}
  \item model fit and implications of posterior
  \end{itemize}
\vfill
\item \myemph{Repeat} as necessary
\end{itemize}
\vfill\hfill {\footnotesize {\slshape Ibid.}}


\sld{Priors, data distributions, and likelihoods}
%
\begin{itemize}
\item We factor joint models into priors and data generating model (more later).
\item The \myemph{prior} is what we know about latent parameters before we see the data.
\item The \myemph{data generating model} generates observable data given parameters.
  \begin{subitemize}
  \item Considered as a function of the parameters for fixed data, the data generating model
    defines the \myemph{likelihood function}.
  \end{subitemize}
\item Both priors and likelihoods are guided by our \myemph{knowledge}. 
\item Both prior and likelihood are \myemph{equally ``subjective''}.
  \begin{subitemize}
  \item Conclusions are typically \myemph{way more sensitive to the likelihood}.
  \item \myemph{``Don't strain at the gnat of prior while swallowing the camel of the
      likelihood.''} (Andrew Gelmans' blog)
  \end{subitemize}
\end{itemize}  

\sld{Where do models come from?}
%
\begin{itemize}
\item Sometimes model comes first, based on substantive
  considerations
\begin{subitemize}
\item toxicology, economics, ecology, physics, \ldots
\end{subitemize}
\item Sometimes model chosen based on data collection
\begin{subitemize}
\item  traditional statistics of surveys and experiments
\end{subitemize}
\item Other times the data comes first
\begin{subitemize}
\item observational studies, meta-analysis, \ldots
\end{subitemize}
\hfill
\item Usually its a mix
\end{itemize}


\sld{Model checking}
%
\begin{itemize}
\item Do the inferences make sense?
\begin{subitemize}
\item are parameter values consistent with model's prior?
\item does simulating from parameter values produce reasonable fake
  data?
\item are marginal predictions consistent with the data?
\end{subitemize}
\item Do predictions and event probabilities for new data make sense?
\vfill
\item {\slshape \myemph{Not}}: Is the model true?
\item {\slshape \myemph{Not}}: What is Pr[model is true]?
\item {\slshape \myemph{Not}}: Can we ``reject'' the model?
\end{itemize}

%
\begin{itemize}
\item Expanding the model

\item more modeling of measurement methods and errors

\end{subitemize}
\item Including more data

\sld{Properties of Bayesian inference}
%
\begin{itemize}
\item Explores full range of parameters consistent with prior info and
  data$^*$
\begin{subitemize}
\item $^*$ if such agreement is possible
\item Stan automates this procedure with diagnostics
\end{subitemize}

\item predictions for future outcomes with uncertainty
\item event probability updates conditioned on data
\item risk assessment / decision analysis conditioned on uncertainty
\end{subitemize}
\end{itemize}

\sld{Model calibration}
%
\begin{itemize}
\item Consider 100 days for which a meteorologist predicted a 70\%
  chance of rain
\begin{subitemize}
\item about 70 of them should have had rain
\item not fewer, not more!
\item technically, expect $\mathsf{Binomial}(100, 0.7)$ rainy day from
  a calibrated model
\end{subitemize}
\item Use posterior predictive checks to test calibration on
\begin{subitemize}
\item training data---can it fit?
\item held out data---can it predict?
\item cross-validation---approximates held out with trainin data
\end{subitemize}
\item Also applies to interval coverage of parameter values
\end{itemize}

\sld{Model sharpness}
%
\begin{itemize}
\item Ideal forecasts are deterministic
\begin{subitemize}
\item predict 100\% chance of rain or 0\% chance of rain
\item always right
\end{subitemize}
%
\item A forecast of 90\% chance of rain reduces uncertainty more than a 50\% prediction
\item A model is \myemph{sharp} if it has narrow posterior intervals
\begin{subitemize}
\item Prediction $\mbox{Pr}[\alpha \in (1.2, 1.9)] = 0.9$
\item is sharper than $\mbox{Pr}[\alpha \in (1, 2)] = 0.9$
\end{subitemize}
\item I.e., sharper models are more certain in its predictions
\item Given calibration, we want our predictions to be \myemph{sharp}
\end{itemize}


\sld{Cross-validation}
%
\begin{itemize}
\item Uses single data set to model held-out performance
\item Assumes stationarity (as most models do)
\item Partition data evenly into disjoint subsets (called \myemph{folds})
\begin{subitemize}
\item 10 is a common choice
\item \myemph{leave-one-out} (LOO) uses a the number of training
  data points
\end{subitemize}
\item For each fold
\begin{subitemize}
\item estimate model on all data but that fold
\item test on that fold
\end{subitemize}
\item Usual comparison statistic is held out log likelihood
\end{itemize}

\sld{Notation for Basic Quantities}
%
\begin{itemize}
\item \myemph{Basic Quantities}
\begin{subitemize}
\item $y$: \ observed data
\item $\theta$: \ parameters (and other unobserved quantities)
\item $x$: \ constants, predictors for conditional (aka ``discriminative'') models
\end{subitemize}
\item \myemph{Basic Predictive Quantities}
\begin{subitemize}
\item $\tilde{y}$: unknown, potentially observable quantities
\item $\tilde{x}$: \ constants, predictors for unknown quantities
\end{subitemize}
\end{itemize}


\sld{Naming Conventions}
%
\begin{itemize}
\item \myemph{Joint}: \ $p(y,\theta)$
\item \myemph{Sampling / Likelihood}: \ $p(y|\theta)$
\begin{subitemize}
\item Sampling is function of $y$ with $\theta$ fixed (prob function)
\item Likelihood is function of $\theta$ with $y$ fixed (\emph{not} prob function)
\end{subitemize}
\item \myemph{Prior}: \ $p(\theta)$
\item \myemph{Posterior}: \ $p(\theta|y)$
\item \myemph{Data Marginal (Evidence)}: \ $p(y)$
\item \myemph{Posterior Predictive}: \ $p(\tilde{y}|y)$
\end{itemize}


\sld{Bayes's Rule for Posterior}
%
\vspace*{-4pt}
\begin{eqnarray*}
    p(\theta|y)
    & = & \frac{p(y,\theta)}{p(y)}
          \hspace*{64pt} \text{\small [def of conditional]}
    \\[6pt]
    & = & \frac{p(y|\theta) \, p(\theta)}{p(y)}
          \hspace*{44pt} \text{\small [chain rule]}
    \\[6pt]
    & = & \frac{p(y|\theta) \, p(\theta)}{\int_{\Theta} p(y,\theta') \ d\theta'}
          \hspace*{40pt} \text{\small [law of total prob]}
    \\[6pt]
    & = & \frac{p(y|\theta) \, p(\theta)}{\int_{\Theta} p(y|\theta') \,
      p(\theta') \ d\theta'}
          \hspace*{20pt} \text{\small [chain rule]}
\end{eqnarray*}

\begin{itemize}
\item \myemph{Inversion:} Final result \myemph{depends only} on
  \myemph{sampling distribution} (likelihood) $p(y|\theta)$ and \myemph{prior}
  $p(\theta)$
\end{itemize}


\sld{Bayes's rule up to proportion}
%
\begin{itemize}
\item If data $y$ is fixed, then
\begin{eqnarray*}
p(\theta|y)
& = & \frac{p(y|\theta) \, p(\theta)}{p(y)}
\\[6pt]
& \propto & p(y|\theta) \, p(\theta)
\\[6pt]
& = & p(y,\theta)
\end{eqnarray*}
\item Posterior proportional to likelihood times prior.
\item Equivalently, posterior proportional to joint.
\vfill
\item Avoids the nasty integral in the denominator for data marginal $p(y)$.
\end{itemize}


\sld{Posterior predictive distribution}
%
\begin{itemize}
\item Predict new data $\tilde{y}$ based on observed data $y$
\item Marginalize parameters $\theta$ out of posterior and likelihood
\begin{eqnarray*}
p(\tilde{y} \mid y)
& = & \mathbb{E}[p(\tilde{y} | \theta) \mid Y = y]
\\[6pt]
& = &
\int p(\tilde{y} | \theta) \, p(\theta | y) \, d\theta.
\end{eqnarray*}
\item Weights predictions $p(\tilde{y}|\theta)$,
by posterior $p(\theta|y)$
\item Integral notation shorthand for sums and/or integrals
\end{itemize}


\sld{Posterior event probabilities}
%
\begin{itemize}
\item Recall that an event $A$ is a collection of outcomes
\item So $A$ may be defined by an indicator $f$ on parameters
\[
f(\theta)
=
\begin{cases}
1 & \text{if } \theta \in A
\\
0 & \text{if } \theta \not\in A
\end{cases}
\]
\begin{subitemize}
\item $f(\theta) = \mathrm{I}(\theta_1 > \theta_2)$
\ \ for \ \ $\Prob{\theta_1 > \theta_2 \, | \, y}$,
\item $f(\theta) = \mathrm{I}(\theta \in (0.50, 0.52)$ for $\mathrm{Pr}\left[ \theta \in (0.50, 0.52) \,|\, y \right]$
\end{subitemize}
\item Defined by posterior expectation of indicator $f(\theta)$
\[
\mathrm{Pr}[A \,|\, y]
\ = \
\mathbb{E}\left[ f(\theta) \, | \, y \right]
\ = \
\int_{\Theta} f(\theta) \, p(\theta|y) \, d\theta.
\]
\end{itemize}


\mypart{Monte Carlo methods}{for solving high-dimensional integrals}

\sld{Monte Carlo methods}
\begin{itemize}
\item Bayesian inference involves \myemph{high-dimensional integrals}.
\item \myemph{Monte Carlo} methods \myemph{estimate} such integrals using probability.
\item Specifically, suppose we want to calculate a \myemph{posterior expectation},
  $$
  \mathbb{E}[f(\theta) \mid y]
  = \int_\Theta f(\theta) \cdot p(\theta \mid y) \textrm{d}y.
  $$
  with respect to a distribution $\theta \sim p(\theta \mid y)$.
\item Suppose we can \myemph{randomly sample} $\drw{\theta}{m} \sim p(\theta \mid y)$
\item Then we can use a plug-in \myemph{Monte Carlo estimate},
  $$
  \mathbb{E}[f(\theta) \mid y]
  \approx \frac{1}{M} \sum_{m=1}^M f(\drw{\theta}{m}).
  $$
\end{itemize}


\sld{Stan samples from the posterior}
%
\begin{itemize}
%
\item Stan's primary mode of operation is \myemph{sampling from a Bayesian posterior}
\item Specifically, Stan will draw a \myemph{posterior sample} of size $M$,
\[
\theta^{(1)}, \ldots, \theta^{(M)} \sim p(\theta \mid y).
\]
\item Stan uses \myemph{Markov chain Monte Carlo}
  \begin{subitemize}
  \item the sequence of draws is not independent
  \item it usually has positive autocorrelation
  \item in simple models, autocorrelation can be negative
  \end{subitemize}
\item Specifically, the \myemph{no-U-turn sampler} (NUTS), a form of \myemph{Hamiltonian Monte Carlo}
  \begin{subitemize}
  \item NUTS/HMC has the best known scaling in dimension among generic samplers.
  \end{subitemize}
\end{itemize}

\sld{Parameter estimation}
\begin{itemize}
\item Parameter estimates are posterior expectations (means)
  \[
\widehat{\theta}
\ = \
\mathbb{E}[\theta \mid y]
\ = \
\int \theta \cdot p(\theta \mid y) \, \mathrm{d}\theta
\ \approx \
\frac{1}{M} \sum_{m=1}^M \theta^{(m)}
\]
%
\begin{subitemize}
\item Posterior mean \myemph{minimizes expected square error}.
\item Posterior median \myemph{minimizes expected absolute error}.
\end{subitemize}
\end{itemize}


\sld{Monte Carlo estimation of $\pi$}

\noindent
\begin{minipage}[t]{0.65\textwidth}
\begin{itemize}
\item Estimating $\pi = 3.14\ldots$ via simulation is \emph{the} textbook application
of Monte Carlo methods.
\vfill
\item Generate points \myemph{uniformly} within the square.
\item \myemph{Area of circle} is: \\
  \emph{proportion within circle} times \emph{square's area}.
\item \myemph{Circle's area} is $\pi;$ \myemph{square's area} is 4.
  \begin{subitemize}
    \item radius $r = 1$ yields area $\pi \cdot r^2 = \pi$

\item \myemph{If} $X, Y \sim \mathsf{Uniform}((-1, 1))$, \myemph{then}
  \vspace*{-4pt}
  \begin{align*}
    4 \cdot \mathrm{Pr}[X^2 + Y^2 \leq 1]
    &= \mathbb{E}\!\left[4 \cdot \textrm{I}[X^2 + Y^2 \leq 1]\right]
    \\
    &= \pi.
    \end{align*}
\end{itemize}
\end{minipage}

\end{itemize}

\sld{Accuracy of Markov chain Monte Carlo}
%
\begin{itemize}
\item Monte Carlo Integration computes the exact posterior to within any $\epsilon$
  \begin{subitemize}
  \item i.e., it is \emph{not} an approximation of the posterior like variational Bayes or a Laplace approxiamtion
    \end{subitemize}
\item Monte Carlo draws are i.i.d. by definition, but MCMC draws are not
\item Central limit theorem extends to MCMC: expected error decreases at rate of $\frac{1}{\sqrt{N}}$. 
\item Need $100 \times$ larger sample for each digit of accuracy
\end{itemize}

\sld{Parameter (co)variance estimation}
%
\begin{itemize}
% \item Given \myemph{posterior mean} estimates $\widehat{\theta}$ as above.
\item Posterior \myemph{parameter variance} estimates for univariate $\theta$ are 
\begin{align*}
\mathrm{var}[\theta \mid y]
\ &= \
    \mathbb{E}\!\left[ (\theta - \widehat{\theta})^2 \right]
\\
\ &\approx \
\frac{1}{M} \sum_{m=1}^M \left(\theta^{(m)} - \widehat{\theta}\,\right)^2
\end{align*}
\item Posterior \myemph{parameter covariance} estimation for multivariate $\theta$ is similar,
\begin{align*}
  \mathrm{cov}[\theta \mid y]
  &= \mathbb{E}\!\left[(\theta - \widehat{\theta}\,) \cdot
      (\theta - \widehat{\theta}\,)^\top\right]
  \\ &\approx
  \frac{1}{M} \sum_{m=1}^M \left(\theta^{(m)} - \widehat{\theta}\,\right)
                          \cdot \left(\theta^{(m)} - \widehat{\theta}\,\right)^\top.
\end{align*}
\end{itemize}

\sld{Posterior event proabilities}
%
\begin{itemize}
\item Posterior \myemph{event probabilities} are computed similarly
\begin{align*}
\mbox{Pr}[\mbox{cond}(\theta) \mid y]
&=
\int \mathrm{I}[\theta \in A] \cdot p(y \mid \theta) \, \mathrm{d}\theta
\\
&\approx
\frac{1}{M} \sum_{m=1}^M \mathrm{I}[\theta^{(m)} \in A].
\end{align*}
%
\item The \myemph{indicator function} $\mathrm{I}[\theta \in A]$ evaluates to 1 if $\theta$ satisfies the event $A$, and evaluates to 0 otherwise.
  \begin{subitemize}
  \item i.e., {\slshape event probabilities are expectations of indicator functions.}
  \end{subitemize}
\end{itemize}

\sld{Posterior predictive densities}
\begin{itemize}
\item Evaluating the \myemph{posterior predictive density} is another expectation,
%
\begin{align*}
p(\tilde{y} \mid y)
& = \mathbb{E}[p(\tilde{y}) \mid y]
\\
&\approx
\frac{1}{M} \sum_{m=1}^M p\left( \tilde{y} \mid \theta^{(m)} \right)
\end{align*}
%
\item \myemph{Sampling} from the posterior predictive requires random number generation,
  \[
    \drw{\tilde{y}}{m} \sim p(\tilde{y} \mid \drw{\theta}{m})
  \]
\item \myemph{Two forms of uncertainty} compound for posterior predictive,
  \begin{subitemize}
  \item \myemph{estimation uncertainty} for $\theta$ given data $y$ represented by $\drw{\theta}{m} \sim p(\theta \mid y)$
  \item \myemph{data generation uncertainty} due to sampling $\drw{tilde{y}}{m} \sim p(\tilde{y} \mid \drw{\theta}{m}$
    \end{subitemize}
\end{itemize}

\sld{Posterior quantiles}
\begin{itemize}
\item Posterior \myemph{quantiles} (e.g., \myemph{intervals} and \myemph{median}) just use the empirical quantiles of the draws,
\[
\mathrm{quantile}(\theta, \alpha)
\ = \
\mathrm{sort}(\drw{\theta}{1:M})
\left[\strut \, \lceil \alpha \times M \rceil \, \right]
\]
\item As $\alpha$ approaches 0 or 1, the number of draws beyond $\alpha$ shrinks and
  \myemph{estimation uncertainty increases dramatically}.
  \begin{subitemize}
  \item e.g., a 2.5\% quantile defining the ends of a central 95\% interval
    has around 2.5 times as much uncertainty (coincidental that it's 2.5 and 2.5\%) as a mean.
  \end{subitemize}
\end{itemize}

\mypart{Generalized linear models}

\sld{Linear regression}
%
\begin{stancode}
data {
  int<lower=0> N;                       // # observations
  vector[N] x;                          // covariates
  vector[N] y;                          // outcomes
}
parameters {
  real alpha;                           // intercept
  real beta;                            // slope
  real<lower=0> sigma;                  // error scale
}
model {
  {alpha, beta} ~ normal(0, 5);         // prior
  sigma ~ lognormal(0, 1);              // prior
  y ~ normal(alpha + beta * x, sigma);  // likelihod
}  
\end{stancode}


\sld{Distribution statements}
%
\begin{itemize}
\item Distribution statements are vectorized, so these two are the same:
\begin{stancode}
  y ~ normal(alpha + beta * x, sigma);   

  for (n in 1:N)
    y[n] ~ normal(alpha + beta * x[n], sigma);
\end{stancode}
%
\item These two target increment statements are also the same as above:
\begin{stancode}  
  target += normal_lupdf(y | alpha + beta * x, sigma);

  for (n in 1:N)
    target += normal_lupdf(y[n] | alpha + beta * x[n], sigma);
\end{stancode}
\end{itemize}

\sld{Stan's constrained parameters}
\begin{itemize}
\item Scale \texttt{sigma} declared \texttt{real<lower=0>} with
  prior \Verb|sigma ~ lognormal(0, 1)|.
  \begin{subitemize}
  \item Stan provides \myemph{uniform} distribution over parameters \myemph{satisfying constraints}.
  \item \myemph{user-defined priors} multiply-in as \myemph{factors}.
  \item $\sigma \in (0, \infty)$ is \myemph{log-transformed} to $\tau = \log \sigma \in (-\infty, \infty)$ for \myemph{sampling}.
  \item adjust for \myemph{change-of-variables} from $\sigma$ to $\tau$
\end{subitemize}
{\small
\vspace*{-4pt}
\begin{align*}
p(\tau)
&= p(\exp(\tau))
   \cdot \left| \frac{\textrm{d}}{\textrm{d}\tau} \exp(\tau) \right|
\\[4pt] &= p(\sigma) \cdot \exp(\tau)
\\[4pt] &= \textrm{lognormal}(\exp(\tau) \mid 0, 1) \cdot \exp(\tau).
\end{align*}
}
\end{itemize}

\sld{Posterior predictive inference}
%
\begin{itemize}
\item Given ``training'' data $(x, y)$ and fresh covariates $x^\textrm{new}$, \myemph{sample}
    from the \myemph{posterior predictive distribution}, 
  $y^{\textrm{new}(m)} \sim p_{Y^\textrm{new} \mid x^\textrm{new}, x, y}$.
\item Include new covariates as \myemph{additional data variables}.
\begin{stancode}  
data {
  ...
  int<lower=0> N_new;     // # predictions
  vector[N_new] x_new;    // covariates for predictions
}
\end{stancode}
\item Use a random number generator in \myemph{generated quantities block}.
\begin{stancode}
generated quantities {
  vector[N_new] y_new = to_vector(normal_rng(alpha + beta * x_new, sigma));
}  
\end{stancode}
\end{itemize}

\sld{Poisson regression}
\begin{itemize}
\item Swap out data declaration and likelihood for Poisson.
\begin{stancode}
data {
  int<lower=0> N;                       // # obs

  array[N] int<lower=0> y;              // count outcomes

  real beta;                            // slope
}

  // note: poisson_log(mu) = poisson(exp(mu))
}  
\end{stancode}
\end{itemize}

\sld{Log odds}
\begin{itemize}
\item \myemph{Logistic regression} is a linear regression on the log odds of a binary outcome.
\item A probability $p \in (0,1) $ has \myemph{odds}  $p:(1 - p)$ and \myemph{odds ratio} $\dfrac{p}{1 - p}$.
\item The \myemph{log odds}, conventionally denoted \emph{logit}, and its inverse,
  $$
  \textrm{logit}(p) = \log \dfrac{p}{1 - p}
  \qquad \qquad
  \textrm{logit}^{-1}(v) = \dfrac{\exp(v)}{1 + \exp(v)}.
  $$
\begin{center}
  \begin{tabular}{c|clr|l}
    $p$ & odds & odds ratio & log odds & note \\ \hline
    1/4 & 1:3 & (1/4) / (3/4) = 1/3 & $- \log 3$ & log odds negative if $p < 0$ \\
    1/2 & 1:1 & (1/2) / (1/2) = 1 & $0$  & \small log odds 0 if $p = 1/2$ \\
    3/4 & 3:1 & (3/4) / (1/4) = 3 & $\log 3$ & \small $\textrm{logit}(p) = -\textrm{logit}(1-p)$ symmetric
  \end{tabular}
\end{center}
\end{itemize}

\sld{Logistic regression}
\begin{itemize}
\item You guessed it---swap for logistic
\end{itemize}
\begin{stancode}
data {
  int<lower=0> N;                       // # obs
  vector[N] x;                          // covariates
  array[N] int<lower=0, upper=1> y;     // binary outcomes
}
parameters {
  real alpha;                           // intercept
  real beta;                            // slope
}
model {
  {alpha, beta} ~ normal(0, 5);           // prior
  y ~ bernoulli_logit(alpha + beta * x);  // log odds: alpha + beta * x
}  
\end{stancode}

\sld{Multiple regression}
\begin{itemize}
\item Data $N \times K$ matrix (w. column of intercepts); slopes $K$-vector.
\end{itemize}
\begin{stancode}
data {
  int<lower=0> N;                       
  int<lower=1> K;              // # covariates per outcome
  matrix[N, K] x;              // data matrix
  array[N] int<lower=0> y;              
}
parameters {
  vector[K] beta;              // K slopes
}
model {
  beta ~ normal(0, 5);         // vectorized prior, following likelihood
  y ~ poisson_log(x * beta);   // vec-mat multiply (x * beta) is N-vector
}  
\end{stancode}

\sld{Autoregressive models}
\begin{itemize}
\item A \myemph{times series} is just a sequence of points $v_1, v_2, \ldots, v_T$.
\item An \myemph{AR(1)} model regresses on the previous point in time for $t > 1$,
  $$
  v_t \sim \textrm{normal}(\alpha + \beta \cdot \alpha_{t-1}, \sigma).
  $$
\item If $\beta \in (-1, 1)$, then as $t \rightarrow \infty$, the \myemph{stationary} distribution is
$$
  v_t \sim \textrm{normal}\!\left(\frac{\alpha}{1 - \beta}, \frac{\sigma}{1 - \beta^2}\right).
  $$
\item An \myemph{AR(k)} model is just a linear regression on the previous k points.
\end{itemize}

\sld{Autorgression in Stan}
\begin{itemize}
\item We can have an autoregressive data distribution or prior; either way
\end{itemize}
\begin{stancode}
vector[T] v;                                          // data or param
  
real alpha; real<lower=0, upper=1> beta;              // stationary param
v[1] ~ normal(alpha / (1 - beta),                     // marginal
              sigma / sqrt(1 - beta^2));              
v[2:T] ~ normal(alpha + beta * v[1:(T - 1)], sigma);  // AR(1)
\end{stancode}
\begin{itemize}
\item Last line is equivalent to
\end{itemize}
\begin{stancode}
for (t in 2:T)
  v[t] ~ normal(alpha + beta * v[t-1], sigma); 
\end{stancode}


\mypart{Crowdsourcing}

\sld{Bradley-Terry crowdsourcing model}
\begin{itemize}
\item Bradley and Terry (early 1950s) collected data on \myemph{consumers comparing
    pairs of products}  to estimate their quality. 
\item Products $j$ are parameterized with a \myemph{real valued quality}
  $\alpha_j \in (-\infty, \infty)$.
\item The $n$-th rating compares \myemph{products} $i_n$ and $j_n$, with \myemph{rating} $y_n = 1$ if product $i_n$ \myemph{preferred} to $j_n$
\item Log odds of $i_n$ being better than $j_n$ is $\alpha_{i[n]} - \alpha_{j[n]}$,
  $$
  y_n \sim \textrm{bernoulli\_logit}(\alpha_{i[n]} - \alpha_{j[n]}).
  $$
\item Elo (1960s) rediscovered a rescaled variant and proposed on online update
  \begin{subitemize}
  \item update is via stochastic gradient descent (Robbins and Monro, early 1950s).
  \end{subitemize}
\end{itemize}

\sld{Bradley-Terry code}
\begin{stancode}
data {
  int<lower=0> N;                       // # matchups
  int<lower-0> J;                       // # items
  array[N] int<lower=1, upper=J> ii;    // first item
  array[N] int<lower=1, upper=J> jj;    // second item
  array[N] int<lower=0, upper=1> y;     // outcomes
}
parameters {
  vector[J] alpha;                      // abilities (log odds scale)
}
model {
  alpha ~ normal(0, 5);                        // prior
  y ~ bernoulli_logit(alpha[ii] - alpha[jj]);  // vectorized likelihood

  // equiv for (n in 1:N) y[n] ~ bernoulli_logit(alpha[ii[n]] - alpha[jj[n]]);
}  
\end{stancode}


\sld{Item-response theory (1PL)}
\begin{itemize}
\item Used for \myemph{educational test design}.
\item Response $y_n$ is 1 if student $j_n$ \myemph{correctly answered} test question $i_n$.
\item Like Bradley-Terry, w.\ student \myemph{abilities} $\alpha_j$ and question \myemph{difficulties} $\beta_i$,
  $$
  y_n \sim \textrm{bernoulli}(\textrm{logit}^{-1}(\alpha_{j[n]} - \beta_{i[n]})).
  $$
\item In Stan,
  \begin{stancode}
    vector[J] alpha;  vector[I] beta; 
    ... 
    y ~ bernoulli_logit(alpha[jj] - beta[ii]); 
  \end{stancode}
\item The \myemph{one-parameter logistic} (1PL) involves difficulty parameter with ability being a ``nuisance'' required to match data.
\end{itemize}

\sld{Item-response theory (2PL)}
\begin{itemize}
\item Want to know how \myemph{discriminative} test items are
  \begin{subitemize}
  \item an item is discriminative if abilities greater than difficulty have high chance of
    getting it correct and abilities lower than difficulty have low chance
  \end{subitemize}
\item Add discrimination $\delta_i > 0$ for items, and use
  $$
  y_n \sim \textrm{bernoulli}(\textrm{logit}^{-1}(\delta_{i[n]} \cdot ((\alpha_{j[n]} - \beta_{i[n]})).
  $$
\item As $\delta_i \rightarrow \infty$, if $\beta_j < \alpha_i$, then log odds go to $-\infty$ and if $\beta_j > \alpha_i$, log odds go to $\infty$.
\end{itemize}
\begin{stancode}
vector[I] delta;
... 
y ~ bernoulli_logit(delta .* (alpha[jj] - beta[ii]));  // .* elementwise
\end{stancode}

\sld{Pooling data (aka sharing strength)}
%
\begin{itemize}
\item \myemph{How do we estimate} with a \myemph{small number of trials}? 
  e.g., 6 hits in 10 at bats, or 0 in 5?
  \begin{subitemize}
  \item maximum likelihood estimates of 0.6 or 0.0 are \myemph{absurd}.
  \end{subitemize}
\item Same logic applies to players with 40 hits in 131 at bats or 152 hits in 537.
\item \myemph{No pooling}: estimate each player separately (assume no similarities)
\item \myemph{Complete pooling}: estimate all players together (assume no differences)
\item \myemph{Partial pooling}: somewhere in the middle (let data decide)
\begin{subitemize}
\item use information about other players (i.e., the population) as warranted by the variation in the data
\end{subitemize}
\end{itemize}


\sld{Complete pooling model}
%
\begin{subitemize}
\item Assume players all have \myemph{same ability} \texttt{phi}.
  \begin{subitemize}
  \item implicit uniform prior 
  \end{subitemize}
\begin{stancode}
data {
  int<lower=0> N;              // items
  array[N] int<lower=0> K;     // trials
  array[N] int<lower=0> y;     // successes
}
parameters {
  real<lower=0, upper=1> phi;  // chance of success
}
model {
  y ~ binomial(K, phi);        // likelihood
}
\end{stancode}
\end{subitemize}


\sld{No pooling model code}
%
\begin{subitemize}
\item Assume each player has independent ability \texttt{phi[n]}.
  \begin{subitemize}
  \item implicit uniform prior 
  \end{subitemize}
\begin{stancode}
data {
  int<lower=0> N;
  array[N] int<lower=0> K;
  array[N] int<lower=0> y;
}
parameters {
  vector<lower=0, upper=1>[N] phi;
}
model {
  y ~ binomial(K, phi);
}
\end{stancode}
\end{subitemize}


\sld{Hierarchical Models}
%
\begin{itemize}
\item Hierarchical models principled way to \myemph{determine how much
    pooling}.
\begin{subitemize}
\item Alternative is to stipulate all or nothing in likelihood (cf. ``subjective'').
\end{subitemize}
\item \myemph{Pull estimates} toward the \myemph{population mean} based on amount of
\myemph{variation} in population
\begin{subitemize}
\item \myemph{low variance} population: more pooling
\item \myemph{high variance} population: less pooling
\end{subitemize}
\item \myemph{In limit}
\begin{subitemize}
\item as variance goes to 0, get complete pooling
\item as variance goes to $\infty$, get no pooling
\end{subitemize}
\end{itemize}


\sld{Hierarchical Batting Ability}
%
\begin{itemize}
\item Instead of fixed priors, \myemph{estimate priors} along with other parameters
\item Still only \myemph{uses data once} for a single model fit (i.e., not empirical Bayes)
\item \myemph{Data}: $y_n, K_n$: hits, at-bats for player $n$
\item \myemph{Parameters}: $\phi_n$: ability for player $n$
\item \myemph{Hyperparameters}: $\alpha, \beta$: population mean and variance
\item \myemph{Hyperpriors}: fixed (constant) priors on $\alpha$ and $\beta$
\end{itemize}


\sld{Hierarchical Batting Model {\small (cont.)}}
%
\begin{eqnarray*}
\theta &  \sim & \distro{Uniform}(0,1)
\\[4pt]
\kappa & \sim & \distro{Pareto}(1.5)
\\[4pt]
\phi_n & \sim & \distro{Beta}(\kappa \, \theta, \ \kappa \, (1 - \theta))
\\[4pt]
y_n & \sim & \distro{Binomial}(K_n, \phi_n)
\end{eqnarray*}
\begin{itemize}
\item $\theta$ is prior mean; \ $\kappa$ is prior count (plus 2). 
\item \myemph{Pareto} provides \myemph{power law} distribution on prior count:
{\small
\[
\distro{Pareto}(u \, | \, \alpha)
\ \propto \
\frac{\alpha}
     {u^{\alpha + 1}}
\]
}
\item Could (\emph{should}!) use more informative prior on hyperparameter $\theta$.
\end{itemize}


\sld{Partial Pooling Model in Stan}
%
\begin{stancode}
data {
  int<lower=0> N;
  array[N] int<lower=0> K;
  array[N] int<lower=0> y;
}
parameters {
  real<lower=0, upper=1> theta;
  real<lower=1> kappa;
  vector<lower=0, upper=1>[N] phi;
}
model {
  kappa ~ pareto(1, 1.5);             // hyperprior
  theta ~ beta(kappa * theta,         // prior
               kappa * (1 - theta));
  y ~ binomial(K, theta);             // likelihood
}
\end{stancode}

\sld{Calibration and Sharpness}
%
\begin{itemize}
\item \myemph{Calibration}:  A model is calibrated if the 50\% intervals
  contain roughly 50\% of the true intervals
\begin{subitemize}
\item technically, we expect $\mbox{Binomial}(N, 0.5)$ of $N$
  parameters to fall in their 50\% intervals
\item we can evaluate with held-out data using cross-validation
\end{subitemize}
%
\item \myemph{Sharpness}: One posterior is sharper than another if it
  concentrates more posterior mass around the true value
\begin{subitemize}
\item e.g., central posterior intervals of interest are narrower
\vfill
\item \footnotesize{see: Gneiting, Balabdaoui, and Raftery (2007) Probabilistic
    forecasts, calibration and sharpness. {\slshape JRSS B}.}
\end{subitemize}
\end{itemize}
